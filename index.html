class LoginPage extends StatefulWidget {

class _LoginPageState extends State {

  // these hold what the user types
  TextEditingController usernameBox = TextEditingController();
  TextEditingController passwordBox = TextEditingController();

  // this shows the spinning circle
  bool isLoading = false;

  // this function runs when you tap Login
  void tapLogin() async {

    // check if boxes are empty
    if (usernameBox.text.trim().isEmpty) {
      showRedMessage("Please type your username");
      return;
    }
    if (passwordBox.text.isEmpty) {
      showRedMessage("Please type your password");
      return;
    }

    // show the spinning circle
    setState(() {
      isLoading = true;
    });

    // ---- REAL PHP API CALL ----
    // we send username and password to the server
    // the server checks if it is correct
    try {
      var serverResponse = await http.post(
        Uri.parse(loginUrl),
        headers: {"Content-Type": "application/json"},
        body: jsonEncode({
          "username": usernameBox.text.trim(),
          "password": passwordBox.text,
        }),
      );

      
      var responseData = jsonDecode(serverResponse.body);

      if (serverResponse.statusCode == 200) {
        myName     = usernameBox.text.trim();
        myUserId   = responseData["user_id"].toString();
        myUserType = responseData["user_type"].toString();
        if (responseData["user_type"] == "User") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => UserHomePage()),
          );

        } else if (responseData["user_type"] == "Hospital") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => HospitalHomePage()),
          );

        } else if (responseData["user_type"] == "Driver") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => DriverHomePage()),
          );

        } else {
          showRedMessage("Unknown user type from server");
        }

      } else {
        // login failed
        showRedMessage(responseData["message"] ?? "Login failed. Try again.");
      }

    } catch (error) {
      // cant reach server
      showRedMessage("Cannot connect to server. Check wifi.");
    }

    // hide the spinning circle
    setState(() {
      isLoading = false;
    });
  }

  TextEditingController nameBox     = TextEditingController();

  TextEditingController phoneBox    = TextEditingController();
  TextEditingController passwordBox = TextEditingController();

  // extra boxes for driver
  TextEditingController licenseBox = TextEditingController();
  TextEditingController vehicleBox = TextEditingController();

  // extra boxes for hospital
  TextEditingController hospNameBox    = TextEditingController();
  TextEditingController imrBox         = TextEditingController();
  TextEditingController hospLocationBox = TextEditingController();
  TextEditingController hospContactBox = TextEditingController();

  // which type is selected  0=User  1=Driver  2=Hospital
  int selectedType = 0;
  List typeNames = ["User", "Driver", "Hospital"];

  bool isLoading = false;

  // this runs when you tap Create Account
  void tapSignup() async {

    // check if basic boxes are empty
    if (nameBox.text.trim().isEmpty ||
        emailBox.text.trim().isEmpty ||
        phoneBox.text.trim().isEmpty ||
        passwordBox.text.isEmpty) {
      showRedMessage("Please fill all the fields");
      return;
    }

    if (passwordBox.text.length < 6) {
      showRedMessage("Password must be at least 6 characters");
      return;
    }

    setState(() {
      isLoading = true;
    });

    // build the data we want to send
    Map dataToSend = {
      "name"      : nameBox.text.trim(),
      "email"     : emailBox.text.trim(),
      "phone"     : phoneBox.text.trim(),
      "password"  : passwordBox.text,
      "user_type" : typeNames[selectedType],
    };

    // add driver extra fields
    if (selectedType == 1) {
      dataToSend["license_no"] = licenseBox.text.trim();
      dataToSend["vehicle_no"] = vehicleBox.text.trim();
    }

    // add hospital extra fields
    if (selectedType == 2) {
      dataToSend["hospital_name"] = hospNameBox.text.trim();
      dataToSend["imr_no"]        = imrBox.text.trim();
      dataToSend["location"]      = hospLocationBox.text.trim();
      dataToSend["contact_no"]    = hospContactBox.text.trim();
    }

    // ---- REAL PHP API CALL ----
    try {
      var serverResponse = await http.post(
        Uri.parse(registerUrl),
        headers: {"Content-Type": "application/json"},
        body: jsonEncode(dataToSend),
      );

      var responseData = jsonDecode(serverResponse.body);

      if (serverResponse.statusCode == 200 || serverResponse.statusCode == 201) {
        // account created!
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Text("Account created! Please login now."),
            backgroundColor: Color(0xFF00C853),
            behavior: SnackBarBehavior.floating,
            shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
          ),
        );
        Navigator.pop(context); // go back to login page

      } else {
        showRedMessage(responseData["message"] ?? "Registration failed. Try again.");
      }

    } catch (error) {
      showRedMessage("Cannot connect to server. Check wifi.");
    }

    setState(() {
      isLoading = false;

class _EmergencyWaitingPageState extends State with SingleTickerProviderStateMixin {

  int secondsLeft = 120;
  Timer? myTimer;
  String statusMessage = "Sending emergency request...";
  bool driverWasFound = false;
  String driverName = "";

  late AnimationController pulseCtrl;
  late Animation pulseAnim;

  @override
  void initState() {
    super.initState();

    // setup the pulsing animation
    pulseCtrl = AnimationController(
      vsync: this,
      duration: Duration(milliseconds: 900),
    );
    pulseCtrl.repeat(reverse: true);
    pulseAnim = Tween(begin: 0.92, end: 1.08).animate(pulseCtrl);

    // start the countdown timer
    startTimer();

    // simulate finding a driver after 6 seconds
    simulateFindingDriver();
  }

  void startTimer() {
    myTimer = Timer.periodic(Duration(seconds: 1), (timer) {
      if (!mounted) return;
      if (secondsLeft <= 0) {
        timer.cancel();
        return;
      }
      setState(() {
        secondsLeft = secondsLeft - 1;
      });
    });
  }

  void simulateFindingDriver() async {
    // wait 2 seconds then show searching message
    await Future.delayed(Duration(seconds: 2));
    if (!mounted) return;
    setState(() { statusMessage = "Notifying nearby drivers..."; });

    // wait 3 more seconds then show expanding message
    await Future.delayed(Duration(seconds: 3));
    if (!mounted) return;
    setState(() { statusMessage = "Round 1: Looking for drivers within 2km..."; });

    // wait 4 more seconds then driver is found!
    await Future.delayed(Duration(seconds: 4));
    if (!mounted) return;

    // driver found! use fake data
    myTimer?.cancel();
    pulseCtrl.stop();
    setState(() {
      driverWasFound = true;
      driverName     = "Ramesh Kumar";
      statusMessage  = "Driver is on the way!";
    });
  }

  @override
  void dispose() {
    myTimer?.cancel();
    pulseCtrl.dispose();
    super.dispose();
  }

    int minutes = secondsLeft ~/ 60;
    int seconds = secondsLeft % 60;
    String mText = minutes.toString().padLeft(2, "0");
    String sText = seconds.toString().padLeft(2, "0");
    return "$mText:$sText";

Line by Line

Function(File) setter
noob This parameter is itself a function! You pass in a mini function that tells _pickImage "where to store the result". Instead of writing 3 separate pick functions for each photo, one function works for all by accepting a setter.
advanced → This is a callback / higher-order function pattern. Function(File) is the type signature — any function that takes a File parameter. Equivalent to passing a lambda as an argument.
XFile? picked
noob XFile is what image_picker returns — a cross-platform file object. The ? means it can be null — it will be null if the user opens the gallery then cancels without picking anything.
advanced → XFile is from the cross_file package (a dependency of image_picker). It abstracts over dart:io File and dart:html for web compatibility. We convert it to a dart:io File via File(picked.path).
imageQuality: 80
noob Compresses the image to 80% quality before storing it. A hospital photo at full quality might be 5–10MB. Compressed to 80% it's typically under 1MB — much faster to upload and easier on your server.
advanced → imageQuality only works for JPEG compression. PNG files are losslessly converted to JPEG first. For medical reports, consider using a higher quality (90-95) to preserve text readability in documents.
if (picked != null)
noob Only runs the code inside if the user actually picked a photo. If they opened the gallery and pressed the back button, picked is null and we do nothing — we don't clear any previously selected photo.
advanced → This is a null guard. Without it, File(null.path) would crash. Dart's null safety would actually flag this at compile time, but explicit null checks keep the logic readable.
setState(() => setter(f))
noob Calls the setter function (which assigns the File to the right variable), wrapped inside setState so Flutter knows to redraw the upload box — showing the image preview instead of the "upload" placeholder.
advanced → Without setState, the File variable updates in memory but the UI stays frozen. setState schedules a build() call. The lambda (f) => _hospitalPhoto = f is a closure that captures the outer variable.

Why MultipartRequest, not http.post()?

💡
http.post() sends only text (JSON or form fields). http.MultipartRequest sends a mixed payload — text fields AND binary file data together. This is how HTML form file uploads work under the hood (multipart/form-data encoding).
request.fields['key'] = value
noob Adds a text value to the request. Like filling out a form on paper — each key is the field name, each value is what you typed. PHP reads these with $_POST['key'].
advanced → fields in MultipartRequest are sent as form-data parts with Content-Disposition: form-data. PHP's $_POST and $_FILES are populated from these parts.
MultipartFile.fromPath('hospital_photo', path)
noob Attaches a file to the request. The first argument 'hospital_photo' is the key — PHP uses $_FILES['hospital_photo'] to access it. The second argument is the file's location on the phone.
advanced → fromPath() reads the file asynchronously (hence await). It auto-detects MIME type from the file extension. You can also pass contentType: MediaType('image','jpeg') explicitly.
response.stream.bytesToString()
noob MultipartRequest returns a StreamedResponse, not a regular Response. You can't read it directly — you must call .stream.bytesToString() to convert the stream into a plain string that jsonDecode can read.
advanced → StreamedResponse streams the body in chunks (efficient for large responses). bytesToString() collects all chunks and joins them. For large file downloads, process the stream chunk-by-chunk instead.

$_POST['key']
noob PHP automatically populates $_POST with all text fields from a multipart form request. Flutter sent request.fields['hospital_name'] = 'City Hospital' → PHP reads it as $_POST['hospital_name'].
advanced → Note: when Flutter sends JSON body (jsonEncode), PHP uses php://input, not $_POST. $_POST is only populated for multipart/form-data or application/x-www-form-urlencoded content types.
password_hash(..., PASSWORD_BCRYPT)
noob NEVER store passwords as plain text. password_hash() scrambles the password into a one-way hash — no one can un-scramble it, not even you. To check a login: use password_verify($input, $storedHash).

uniqid('hosp_', true)

advanced → PDO prepared statements send the query structure and data separately to the DB server. The DB handles escaping. This is fundamentally safer than string concatenation (even with mysqli_real_escape_string).
