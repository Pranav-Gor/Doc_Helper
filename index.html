class LoginPage extends StatefulWidget {

class _LoginPageState extends State {

  // these hold what the user types
  TextEditingController usernameBox = TextEditingController();
  TextEditingController passwordBox = TextEditingController();

  // this shows the spinning circle
  bool isLoading = false;

  // this function runs when you tap Login
  void tapLogin() async {

    // check if boxes are empty
    if (usernameBox.text.trim().isEmpty) {
      showRedMessage("Please type your username");
      return;
    }
    if (passwordBox.text.isEmpty) {
      showRedMessage("Please type your password");
      return;
    }

    // show the spinning circle
    setState(() {
      isLoading = true;
    });

    // ---- REAL PHP API CALL ----
    // we send username and password to the server
    // the server checks if it is correct
    try {
      var serverResponse = await http.post(
        Uri.parse(loginUrl),
        headers: {"Content-Type": "application/json"},
        body: jsonEncode({
          "username": usernameBox.text.trim(),
          "password": passwordBox.text,
        }),
      );

      
      var responseData = jsonDecode(serverResponse.body);

      if (serverResponse.statusCode == 200) {
        myName     = usernameBox.text.trim();
        myUserId   = responseData["user_id"].toString();
        myUserType = responseData["user_type"].toString();
        if (responseData["user_type"] == "User") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => UserHomePage()),
          );

        } else if (responseData["user_type"] == "Hospital") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => HospitalHomePage()),
          );

        } else if (responseData["user_type"] == "Driver") {
          Navigator.pushReplacement(
            context,
            MaterialPageRoute(builder: (context) => DriverHomePage()),
          );

        } else {
          showRedMessage("Unknown user type from server");
        }

      } else {
        // login failed
        showRedMessage(responseData["message"] ?? "Login failed. Try again.");
      }

    } catch (error) {
      // cant reach server
      showRedMessage("Cannot connect to server. Check wifi.");
    }

    // hide the spinning circle
    setState(() {
      isLoading = false;
    });
  }

  TextEditingController nameBox     = TextEditingController();

  TextEditingController phoneBox    = TextEditingController();
  TextEditingController passwordBox = TextEditingController();

  // extra boxes for driver
  TextEditingController licenseBox = TextEditingController();
  TextEditingController vehicleBox = TextEditingController();

  // extra boxes for hospital
  TextEditingController hospNameBox    = TextEditingController();
  TextEditingController imrBox         = TextEditingController();
  TextEditingController hospLocationBox = TextEditingController();
  TextEditingController hospContactBox = TextEditingController();

  // which type is selected  0=User  1=Driver  2=Hospital
  int selectedType = 0;
  List typeNames = ["User", "Driver", "Hospital"];

  bool isLoading = false;

  // this runs when you tap Create Account
  void tapSignup() async {

    // check if basic boxes are empty
    if (nameBox.text.trim().isEmpty ||
        emailBox.text.trim().isEmpty ||
        phoneBox.text.trim().isEmpty ||
        passwordBox.text.isEmpty) {
      showRedMessage("Please fill all the fields");
      return;
    }

    if (passwordBox.text.length < 6) {
      showRedMessage("Password must be at least 6 characters");
      return;
    }

    setState(() {
      isLoading = true;
    });

    // build the data we want to send
    Map dataToSend = {
      "name"      : nameBox.text.trim(),
      "email"     : emailBox.text.trim(),
      "phone"     : phoneBox.text.trim(),
      "password"  : passwordBox.text,
      "user_type" : typeNames[selectedType],
    };

    // add driver extra fields
    if (selectedType == 1) {
      dataToSend["license_no"] = licenseBox.text.trim();
      dataToSend["vehicle_no"] = vehicleBox.text.trim();
    }

    // add hospital extra fields
    if (selectedType == 2) {
      dataToSend["hospital_name"] = hospNameBox.text.trim();
      dataToSend["imr_no"]        = imrBox.text.trim();
      dataToSend["location"]      = hospLocationBox.text.trim();
      dataToSend["contact_no"]    = hospContactBox.text.trim();
    }

    // ---- REAL PHP API CALL ----
    try {
      var serverResponse = await http.post(
        Uri.parse(registerUrl),
        headers: {"Content-Type": "application/json"},
        body: jsonEncode(dataToSend),
      );

      var responseData = jsonDecode(serverResponse.body);

      if (serverResponse.statusCode == 200 || serverResponse.statusCode == 201) {
        // account created!
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Text("Account created! Please login now."),
            backgroundColor: Color(0xFF00C853),
            behavior: SnackBarBehavior.floating,
            shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
          ),
        );
        Navigator.pop(context); // go back to login page

      } else {
        showRedMessage(responseData["message"] ?? "Registration failed. Try again.");
      }

    } catch (error) {
      showRedMessage("Cannot connect to server. Check wifi.");
    }

    setState(() {
      isLoading = false;

class _EmergencyWaitingPageState extends State with SingleTickerProviderStateMixin {

  int secondsLeft = 120;
  Timer? myTimer;
  String statusMessage = "Sending emergency request...";
  bool driverWasFound = false;
  String driverName = "";

  late AnimationController pulseCtrl;
  late Animation pulseAnim;

  @override
  void initState() {
    super.initState();

    // setup the pulsing animation
    pulseCtrl = AnimationController(
      vsync: this,
      duration: Duration(milliseconds: 900),
    );
    pulseCtrl.repeat(reverse: true);
    pulseAnim = Tween(begin: 0.92, end: 1.08).animate(pulseCtrl);

    // start the countdown timer
    startTimer();

    // simulate finding a driver after 6 seconds
    simulateFindingDriver();
  }

  void startTimer() {
    myTimer = Timer.periodic(Duration(seconds: 1), (timer) {
      if (!mounted) return;
      if (secondsLeft <= 0) {
        timer.cancel();
        return;
      }
      setState(() {
        secondsLeft = secondsLeft - 1;
      });
    });
  }

  void simulateFindingDriver() async {
    // wait 2 seconds then show searching message
    await Future.delayed(Duration(seconds: 2));
    if (!mounted) return;
    setState(() { statusMessage = "Notifying nearby drivers..."; });

    // wait 3 more seconds then show expanding message
    await Future.delayed(Duration(seconds: 3));
    if (!mounted) return;
    setState(() { statusMessage = "Round 1: Looking for drivers within 2km..."; });

    // wait 4 more seconds then driver is found!
    await Future.delayed(Duration(seconds: 4));
    if (!mounted) return;

    // driver found! use fake data
    myTimer?.cancel();
    pulseCtrl.stop();
    setState(() {
      driverWasFound = true;
      driverName     = "Ramesh Kumar";
      statusMessage  = "Driver is on the way!";
    });
  }

  @override
  void dispose() {
    myTimer?.cancel();
    pulseCtrl.dispose();
    super.dispose();
  }

    int minutes = secondsLeft ~/ 60;
    int seconds = secondsLeft % 60;
    String mText = minutes.toString().padLeft(2, "0");
    String sText = seconds.toString().padLeft(2, "0");
    return "$mText:$sText";
